	if(a>b)
		disp('Holi, dentro del if c:');
	elseif(a<b)
		disp('Holi, dentro del elseif c:');
	else
		disp('Holi, dentro del else c:');
	end %Este end representa el termino del bloque if (el cual incluye a else)
